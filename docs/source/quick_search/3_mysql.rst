.. code-block:: bash

    # 注意替换变量
    # 导出数据库${database}到${file}
    mysqldump -h ${host} -P ${port} -u ${user} -p${passwd} ${database} >${file}
    # 相应的导入
    mysql -u ${user} -p${passwd} ${database} <${file}
